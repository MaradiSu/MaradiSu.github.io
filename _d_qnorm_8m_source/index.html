3 % DQMULT Dual quaternion multiplication
4 %
5 % DS = DQMULT(DQ,DR) returns the dual quaternion DS which is the dual
6 % quaternion product of DQ and DR. DQ (resp. DR, DS) is a vector of
7 % size 8 representing a dual quaternion or an array of size 8*N (column
8 % i represents dual quaternion i) where N is the number of dual
9 % quaternions.
10 % There are several combinations possible:
11 % - DQ is a 8-vector and DR is an array 8*N ==> DS is an array 8*N.
12 % DQ is multiplied with each column (dual quaternion) of DR.
13 % - DQ is an array 8*N and DR is a 8-vector ==> DS is an array 8*N.
14 % Each column (dual quaternion) of DQ is multiplied with DR.
15 % - DQ and DR are both arrays of size 8*N ==> DS is an array 8*N.
16 % The dual quaternion muliplication is carried out on corresponding
17 % columns.
18 % THE ORDER MATTERS: the dual quaternion muliplication is NOT
19 % commutative: DQMULT(Q,R) is different from DQMULT(R,Q)
20 %
21 % DS = DQMULT(DQ1,DQ2,DQ3,...,DQN) returns the dual quaternion DS which
22 % is the dual quaternion product of DQ1, DQ2, ... and DQN.
23 % If * represents the dual quaternion mulitplication, then we can
24 % write: DS = DQ1*DQ2*DQ3*...*DQN. Please note that the dual
25 % quaternion product is associative: (DQ1*DQ2)*DQ3) = DQ1*(DQ2*DQ3) =
26 % DQ1*DQ2*DQ3.
27 % DQ1, DQ2, ..., DQR must have the same format as above (see DS =
28 % DQMULT(DQ,DR))
29 %
30 % See also QMULT, DQCONJ
31 
32 soptargin = size(varargin,2); % number of optional dual quaternions to multiply
33 sdq = size(dq);
34 sdr = size(dr);
35 if sdq == [1 8], dq = dq'; sdq = size(dq); end
36 if sdr == [1 8], dr = dr'; sdr = size(dr); end
37 
38 if soptargin > 0 % more than 2 dual quaternions: use of recursion
39  temp = DQmult(dq,dr);
40  dqopt1 = varargin{1,1};
41  sdqopt1 = size(dqopt1);
42  if sdqopt1 == [1 8], dqopt1 = dqopt1'; end
43  ds = DQmult(temp,dqopt1,varargin{1,2:end});
44 else % dual quaternion multiplication of two dual quaternions
45 
46  % wrong size
47  if sdq(1) ~= 8 || sdr(1) ~= 8
48  error('DualQuaternion:DQmult:wrongsize',...
49  '%d rows in array dq and %d rows in array dr. It should be 8 for both.',...
50  sdq(1),sdr(1));
51  end
52 
53  nq = sdq(2);
54  nr = sdr(2);
55  % wrong format for the inputs
56  if nq ~= nr && nq > 1 && nr > 1
57  error('DualQuaternion:DQmult:wrongFormat',...
58  '%d columns in array q and %d columns in array r. It should be the same number for both, or one of these should be 1.',...
59  nq,nr);
60  end
61

64  ds1 = Qmult(dq(1:4,:),dr(5:8,:))+Qmult(dq(5:8,:),dr(1:4,:));
65  ds = [ds0 ; ds1];
66 end
