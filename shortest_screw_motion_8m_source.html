3 % SHORTESTSCREWMOTION returns the smallest screw motion from line L1 to
4 % line L2
5 %
6 % [THETA,D,AXIS,AXISPOINT] = SHORTESTSCREWMOTION(L1,L2) returns the
7 % parameters of the shortest (the smallest THETA and D among all
8 % possible solutions) screw motion from dual quaternion line L1
9 % to dual quaternion line L2.
10 % - L1 (resp. L2) is a line position dual quaternion. It is a 8-vector
11 % or a 8*N array (column i represents line position dual
12 % quaternion i) where N is the number of lines in array L1 (resp.
13 % L2). See also LINE2DQUAT.
14 % - THETA is the rotation angle [deg] around the screw motion axis
15 % AXIS. It is a 1*N array.
16 % - D is the distance along the screw motion axis AXIS. It is a 1*N
17 % array.
18 % - AXIS is the unitary screw motion axis. It is a 3*N array.
19 % - AXISPOINT is the point of the screw motion axis line which is at
20 % mid-distance between lines L1 and L2. If L1 and L2 intersect,
21 % AXISPOINT is the intersection point of the two lines.
22 %
23 % [THETA,D,AXIS,AXISPOINT] = SHORTESTSCREWMOTION(L1,L2,OPTION) returns
24 % the same parameters but the point of the screw motion axis can be
25 % specified by OPTION. If OPTION is:
26 % - 'orthogonal': AXISPOINT is the point of the screw motion axis
27 % line which is orthogonal to the screw motion axis AXIS (It
28 % is the point of the screw motion axis line the closest
29 % from the origin).
30 % - 'intersection' (default): AXISPOINT is the point of the screw
31 % motion axis line which is at mid-distance between lines L1
32 % and L2. If L1 and L2 intersect, AXISPOINT is the
33 % intersection point of the two lines.
34 %
35 % DQ = SHORTESTSCREWMOTION(L1,L2) returns the screw motion dual
36 % quaternion which represents the smallest screw motion between
37 % lines L1 and L2.
38 % - DQ is a screw motion dual quaternion. It is a 8*N array.
39 %
40 % See also DQUAT2SCREW, SHORTESTROTATION, SCREW2DQUAT, LINE2DQUAT
41 
42 sL1 = size(L1);
43 sL2 = size(L2);
44 if sL1 == [1 3], L1 = L1'; sL1 = size(L1); end
45 if sL2 == [1 3], L2 = L2'; sL2 = size(L2); end

48 if sL1(1) ~= 8 || sL2(1) ~= 8
49  error('DualQuaternion:shortestScrewMotion:wrongsize',...
50  '%d rows in array L1 and %d rows in array L2. It should be 8 for both.',...
51  sL1(1),sL2(1));
52 end
53 
54 np1 = sL1(2);
55 np2 = sL2(2);
56 if np1 ~= np2

62 dqt = DQmult(DQconj(L1,'line'),L2);

68 sintheta = sind(theta);
69 indthetaOK = find(sintheta ~= 0);
70 ind0 = find(sintheta == 0);
71 
72 % d
73 d = zeros(1,np1);
74 if ~isempty(indthetaOK)
75  d(indthetaOK) = -dqt(5,indthetaOK)./sintheta(indthetaOK);
76 end
77 d(ind0) = sqrt(sum(dqt(6:8,ind0).^2));
78 inddOK = find(d>0);
79

85 end
86 ind2 = intersect(inddOK,ind0);

97  'The specified option is not correctly encoded. It must be ''orthogonal'' or ''intersection''. ');
98  end
99 elseif soptargin > 1

101  'Too many inputs');
102 end
103 
104 axispoint = zeros(3,np1);
105 m_axis = zeros(3,np1);
106 repcos = repmat(costheta(indthetaOK),3,1);
107 repd = repmat(d(indthetaOK),3,1);
108 if ~isempty(indthetaOK)
109  m_axis(:,indthetaOK) = (dqt(6:8,indthetaOK)-repd.*repcos.*axis(:,indthetaOK))./repsin;
110  axispoint(:,indthetaOK) = cross(axis(:,indthetaOK),m_axis(:,indthetaOK)); % gives the point orthogonal to the point orientation
111  if strcmp(choice_ascrew,'intersection')

116  end
117 end
