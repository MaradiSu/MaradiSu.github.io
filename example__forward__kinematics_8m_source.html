1 %% EXAMPLE_FORWARD_KINEMATICS
2 % EXAMPLE_FORWARD_KINEMATICS describes the forward kinematics for 2 or
3 % 3-link arms. For both types, the user can choose to use
4 % rotations and translation dual quaternions, or to use screw
5 % motion dual quaternions. For the two-link arm, only the
6 % end-effector position fporward kinematics is computed, while
7 % for the 3-link arm, the end-effector orientation and position
8 % forward kinematics are computed.
9 % The user must specify the arm parameters at the beginning of
10 % this file. Two parameters are important:
11 % - CHOICE_ARM: scalar value equal to:
12 % --> 2 : the two-link arm is chosen
13 % --> 3 : the three-link arm is chosen
14 % - CHOICE_REPRESENTATION: scalar value equal to:
15 % --> 1 : only rotations and translations are considered
16 % --> 2 : screw motions are considered
17 % Depending on the option you choose, the end-effector point
18 % (and orientation) position and velocity is displayed.
19 
20 
21 %% parameters to be specified by the user
22 clear all;
23 close all;
24 clc;
25 
26 choice_arm = 3; % 2 = two-link arm, 3 = three-link arm
27 choice_representation = 1; % 1 = rotation and translations, 2 = screw motions
28 % arm segement parameters
29 a = 10; % length of the upper arm
30 b = 8; % length of the lower arm
31 c = 2; % distance between the wrist and the end-effector (e.g. a fingertip, or a tool)
32 
33 % rotation parameters for the upper arm rotation around the shoulder
34 theta_UB = 0; % rotation angle [deg] of the upper arm around the shoulder
35 axis_UB = [0 0 1]'; % rotation axis of the upper arm around the shoulder
36 Omega_speed_UB = 0; % instantaneous rotational speed [deg/s] of the upper arm around the shoulder
37 Omega_axis_UB = [0 0 -1]'; % instantaneous rotational velocity axis of the upper arm around the shoulder
38 
39 % rotation parameters for the lower arm rotation around the elbow
40 theta_LU = 90; % rotation angle [deg] of the lower arm around the elbow
41 axis_LU = [0 0 1]'; % rotation axis of the lower arm around the elbow
42 Omega_speed_LU = 20; % instantaneous rotational speed [deg/s] of the lower arm around the elbow
43 Omega_axis_LU = [0 0 -1]'; % instantaneous rotational velocity axis of the lower arm around the elbow arm
44 
45 if choice_arm == 3
46  % rotation parameters for the hand rotation around the wrist
47  theta_HL = 90; % rotation angle [deg] of the hand around the wrist
48  axis_HL = [0 0 1]'; % rotation axis of the hand around the wrist
49  Omega_speed_HL = 0; % instantaneous rotational speed [deg/s] of the hand around the wrist
50  Omega_axis_HL = [0 0 -1]'; % instantaneous rotational velocity axis of the hand around the wrist
51 end
52 
53 %% Forward kinematics of the end-effector of a two-link arm (point transformation)
54 if choice_arm == 2 % the two-link arm is considered
55 %% Using rotations and translations
56  if choice_representation == 1 % rotations and translations operators
57  % rotation, translation and rotational velocities dual quaternions
58  R_UB = rot2dquat(theta_UB,axis_UB); % rotation dual quaternion of the upper arm around the shoulder
59  R_LU = rot2dquat(theta_LU,axis_LU); % rotation dual quaternion of the lower arm around the elbow
60  T_ES = trans2dquat([1 0 0]',a); % translation dual quaternion specifying the offset between the elbow and the shoulder
61  % (in upper-arm fixed shoulder centered
62  % coordinates)
63  Omega_UB = rotvel2dquat(Omega_speed_UB,Omega_axis_UB); % rotational velocity dual quaternion of the upper arm around the shoulder
64  Omega_LU = rotvel2dquat(Omega_speed_LU,Omega_axis_LU); % rotational velocity dual quaternion of the lower arm around the elbow
65 
66  % position
67  P_LE = [b 0 0]'; % position of the end-effector in (lower arm)-fixed elbow-centered coordinates
68  P_LE_dq = pos2dquat(P_LE); % position dual quaternion for P_LE
69  S_tot = DQmult(R_LU,T_ES,R_UB);
70  P_BS_dq = DQmult(DQconj(S_tot),P_LE_dq,S_tot); % Forward kinematics computation
71  P_BS = dquat2pos(P_BS_dq);
72 
73  % velocity (see equation (18) in the article)
74  P_UE_dq = DQmult(DQconj(R_LU),P_LE_dq,R_LU); % position of the end-effector in (upper arm)-fixed elbow centered coordinates
75  antisym_LU = 0.5*(DQmult(P_UE_dq,Omega_LU)-DQmult(Omega_LU,P_UE_dq));
76  antisym_UB = 0.5*(DQmult(P_BS_dq,Omega_UB)-DQmult(Omega_UB,P_BS_dq));
77  vel_P_BS_dq = antisym_UB + DQmult(DQconj(R_UB),antisym_LU,R_UB);
78  vel_P_BS = dquat2vel(vel_P_BS_dq);
79 %% Using screw motions
80  elseif choice_representation == 2% screw motion representation: the used reference frame is shoulder-centered body-fixed
81  % parameters of the screw motion of the lower arm around the elbow
82  d_LU = 0; % translation distance along the screw motion axis
83  axispoint_LU = [a 0 0]'; % offset between the screw motion axis line and the origin (in the reference configuration)
84  S_LU = screw2dquat(theta_LU,d_LU,axis_LU,axispoint_LU); % screw motion dual quaternion of the lower arm around the elbow.
85  % obviously, there is no translation component along the screw axis.
86  S_LU_dot = screwvel2dquat(Omega_speed_LU,Omega_axis_LU,0,[0 0 0],S_LU); % screw motion velocity dual quaternion (due to the rotational speed only)
87 
88  % parameters of the screw motion of the upper arm around the shoulder
89  d_UB = 0;
90  axispoint_UB = [0 0 0]';
91  S_UB = screw2dquat(theta_UB,d_UB,axis_UB,axispoint_UB);
92  S_UB_dot = screwvel2dquat(Omega_speed_UB,Omega_axis_UB,0,[0 0 0],S_UB);
93 
94  % position
95  P_BS_0 = [a+b 0 0]'; % reference position (arm extended) of the end-effector (in body fixed shoulder centered coordinates)
96  P_BS_0_dq = pos2dquat(P_BS_0);
97  S_tot = DQmult(S_LU,S_UB);
98  P_BS_dq = DQmult(DQconj(S_tot),P_BS_0_dq,S_tot); % Forward kinematics computation
99  P_BS = dquat2pos(P_BS_dq);
100 
101  % velocity
102  P_elbow_dq = DQmult(DQconj(S_LU),P_BS_0_dq,S_LU);
103  comp_UB_dot = DQmult(DQconj(S_UB_dot),P_elbow_dq,S_UB)+DQmult(DQconj(S_UB),P_elbow_dq,S_UB_dot);
104  comp_LU_dot = DQmult(DQconj(S_LU_dot),P_BS_0_dq,S_LU)+DQmult(DQconj(S_LU),P_BS_0_dq,S_LU_dot);
105  vel_P_BS_dq = comp_UB_dot+DQmult(DQconj(S_UB),comp_LU_dot,S_UB);
106  vel_P_BS = dquat2vel(vel_P_BS_dq);
107  end
108  disp('The two-link arm end-effector position (in body-fixed shoulder-centered coordinates) is :');
109  disp(P_BS);
110  disp('The two-link arm end-effector velocity (in body-fixed shoulder-centered coordinates) is :');
111  disp(vel_P_BS);
112 
113  %% Forward kinematics of the end-effector of a three-link arm (point and line transformations)
114 elseif choice_arm == 3 % the three-link arm is considered
115  %% Using rotations and translations
116  if choice_representation == 1 % rotations and translations operators
117  % rotation, translation and rotational velocities dual quaternions
118  R_UB = rot2dquat(theta_UB,axis_UB); % rotation dual quaternion of the upper arm around the shoulder
119  R_LU = rot2dquat(theta_LU,axis_LU); % rotation dual quaternion of the lower arm around the elbow
120  R_HL = rot2dquat(theta_HL,axis_HL); % rotation dual quaternion of the hand around the wrist
121  T_ES = trans2dquat([1 0 0]',a); % translation dual quaternion specifying the offset between the elbow and the shoulder
122  T_WE = trans2dquat([1 0 0]',b); % translation dual quaternion specifying the offset between the wrist and the elbow
123 
124  Omega_UB = rotvel2dquat(Omega_speed_UB,Omega_axis_UB); % rotational velocity dual quaternion of the upper arm around the shoulder
125  Omega_LU = rotvel2dquat(Omega_speed_LU,Omega_axis_LU); % rotational velocity dual quaternion of the lower arm around the elbow
126  Omega_HL = rotvel2dquat(Omega_speed_HL,Omega_axis_HL); % rotational velocity dual quaternion of the hand around the wrist
127 
128  % position
129  orientation_tool_H = [0 1 0]'; % tool (end-effector) orientation in hand-fixed coordinates
130  position_tool_HW = [c 0 0]'; % tool position in hand-fixed wrist-centered coordinates
131  L_HW_dq = line2dquat(orientation_tool_H,position_tool_HW); % line position dual quaternion in hand-fixed wrist-centered coordinates
132  P_HW_dq = pos2dquat(position_tool_HW); % position dual quaternion specifying the tool position in hand-fixed wrist-centered coordinates
133  S_tot = DQmult(R_HL,T_WE,R_LU,T_ES,R_UB);
134  L_BS_dq = DQmult(DQconj(S_tot,'line'),L_HW_dq,S_tot); % Forward kinematics computation for the line position
135  P_BS_dq = DQmult(DQconj(S_tot),P_HW_dq,S_tot);
136  [orientation_tool_B,r] = dquat2line(L_BS_dq);
137  P_BS = dquat2pos(P_BS_dq);
138 
139  % velocity
140  orientation_tool_H_dot = [0 0 0]'; % the tool orientation rate of change (in hand-fixed coordinates)
141  position_tool_H_dot = [0 0 0]'; % the tool velocity (in hand-fixed coordinates)
142  L_HW_dot = linevel2dquat(orientation_tool_H,position_tool_HW,orientation_tool_H_dot,position_tool_H_dot);
143  P_HW_dot_dq = vel2dquat(position_tool_H_dot);
144 
145  % end-effector orientation (line transfo)
146  T_WE_R_LU = DQmult(T_WE,R_LU);
147  L_LW = DQmult(DQconj(R_HL),L_HW_dq,R_HL);
148  L_UE = DQmult(DQconj(T_WE_R_LU,'line'),L_LW,T_WE_R_LU);
149  antisym_UB = 0.5*(DQmult(L_BS_dq,Omega_UB)-DQmult(Omega_UB,L_BS_dq));
150  antisym_LU = 0.5*(DQmult(L_UE,Omega_LU)-DQmult(Omega_LU,L_UE));
151  antisym_HL = 0.5*(DQmult(L_LW,Omega_HL)-DQmult(Omega_HL,L_LW));
152  R_LU_R_UB = DQmult(R_LU,R_UB);
153  R_tot = DQmult(R_HL,R_LU,R_UB);
154  L_BS_dot = antisym_UB+...
155  DQmult(DQconj(R_UB),antisym_LU,R_UB)+...
156  DQmult(DQconj(R_LU_R_UB),antisym_HL,R_LU_R_UB)+...
157  DQmult(DQconj(R_tot),L_HW_dot,R_tot);
158  [orientation_tool_B_dot,MP] = dquat2linevel(L_BS_dot) ;
159 
160  % end-effector position (point transfo)
161  P_LW = DQmult(DQconj(R_HL),P_HW_dq,R_HL);
162  P_UE = DQmult(DQconj(T_WE_R_LU),P_LW,T_WE_R_LU);
163  antisym_P_UB = 0.5*(DQmult(P_BS_dq,Omega_UB)-DQmult(Omega_UB,P_BS_dq));
164  antisym_P_LU = 0.5*(DQmult(P_UE,Omega_LU)-DQmult(Omega_LU,P_UE));
165  antisym_P_HL = 0.5*(DQmult(P_LW,Omega_HL)-DQmult(Omega_HL,P_LW));
166  P_BS_dot_dq = antisym_P_UB+...
167  DQmult(DQconj(R_UB),antisym_P_LU,R_UB)+...
168  DQmult(DQconj(R_LU_R_UB),antisym_P_HL,R_LU_R_UB)+...
169  DQmult(DQconj(R_tot),P_HW_dot_dq,R_tot);
170  P_BS_dot = dquat2vel(P_BS_dot_dq);
171  %% Using screw motions
172  elseif choice_representation == 2% screw motion representation: the used reference frame is shoulder-centered body-fixed
173  % parameters of the screw motion of the lower arm around the elbow
174  d_HL = 0; % translation distance along the screw motion axis
175  axispoint_HL = [a+b 0 0]'; % offset between the screw motion axis line and the origin (in the reference configuration)
176  S_HL = screw2dquat(theta_HL,d_HL,axis_HL,axispoint_HL); % screw motion dual quaternion of the hand around the wrist.
177  % obviously, there is no translation component along the screw axis.
178  S_HL_dot = screwvel2dquat(Omega_speed_HL,Omega_axis_HL,0,[0 0 0],S_HL); % screw motion velocity dual quaternion (due to the rotational speed only)
179 
180  % parameters of the screw motion of the lower arm around the elbow
181  d_LU = 0; % translation distance along the screw motion axis
182  axispoint_LU = [a 0 0]'; % offset between the screw motion axis line and the origin (in the reference configuration)
183  S_LU = screw2dquat(theta_LU,d_LU,axis_LU,axispoint_LU); % screw motion dual quaternion of the lower arm around the elbow.
184  % obviously, there is no translation component along the screw axis.
185  S_LU_dot = screwvel2dquat(Omega_speed_LU,Omega_axis_LU,0,[0 0 0],S_LU); % screw motion velocity dual quaternion (due to the rotational speed only)
186 
187  % parameters of the screw motion of the upper arm around the shoulder
188  d_UB = 0;
189  axispoint_UB = [0 0 0]';
190  S_UB = screw2dquat(theta_UB,d_UB,axis_UB,axispoint_UB);
191  S_UB_dot = screwvel2dquat(Omega_speed_UB,Omega_axis_UB,0,[0 0 0],S_UB);
192 
193  % position
194  orientation_tool_0 = [0 1 0]'; % reference orientation of the end-effector (in body fixed shoulder centered coordinates)
195  position_tool_0 = [a+b+c 0 0]'; % reference position (arm extended) of the end-effector (in body fixed shoulder centered coordinates)
196  P_BS_0_dq = pos2dquat(position_tool_0);
197  L_BS_0 = line2dquat(orientation_tool_0,position_tool_0);
198  S_tot = DQmult(S_HL,S_LU,S_UB);
199  L_BS = DQmult(DQconj(S_tot,'line'),L_BS_0,S_tot); % Forward kinematics computation for the line position
200  P_BS_dq = DQmult(DQconj(S_tot),P_BS_0_dq,S_tot);
201  [orientation_tool_B,r] = dquat2line(L_BS);
202  P_BS = dquat2pos(P_BS_dq);
203 
204  % velocity
205  orientation_tool_0_dot = [0 0 0]'; % the reference orientation rate of change
206  position_tool_0_dot = [0 0 0]'; % the reference tool velocity
207  L_BS_0_dot = linevel2dquat(orientation_tool_0,position_tool_0,orientation_tool_0_dot,position_tool_0_dot);
208 
209  % end-effector orientation (line transfo)
210  L_w = DQmult(DQconj(S_HL,'line'),L_BS_0,S_HL);
211  L_e = DQmult(DQconj(S_LU,'line'),L_w,S_LU);
212  comp_HL_dot = DQmult(DQconj(S_HL_dot,'line'),L_BS_0,S_HL)+DQmult(DQconj(S_HL,'line'),L_BS_0,S_HL_dot);
213  comp_LU_dot = DQmult(DQconj(S_LU_dot,'line'),L_w,S_LU)+DQmult(DQconj(S_LU,'line'),L_w,S_LU_dot);
214  comp_UB_dot = DQmult(DQconj(S_UB_dot,'line'),L_e,S_UB)+DQmult(DQconj(S_UB,'line'),L_e,S_UB_dot);
215  S_LU_S_UB = DQmult(S_LU,S_UB);
216  L_BS_dot = DQmult(DQconj(S_tot,'line'),L_BS_0_dot,S_tot)+...
217  DQmult(DQconj(S_LU_S_UB,'line'),comp_HL_dot,S_LU_S_UB)+...
218  DQmult(DQconj(S_UB,'line'),comp_LU_dot,S_LU_S_UB)+...
219  comp_UB_dot;
220  [orientation_tool_B_dot,MP] = dquat2linevel(L_BS_dot);
221 
222  % end-effector position (point transfo)
223  P_BS_0_dot_dq = vel2dquat(position_tool_0_dot);
224  P_w = DQmult(DQconj(S_HL),P_BS_0_dq,S_HL);
225  P_e = DQmult(DQconj(S_LU),P_w,S_LU);
226  comp_P_HL_dot = DQmult(DQconj(S_HL_dot),P_BS_0_dq,S_HL)+DQmult(DQconj(S_HL),P_BS_0_dq,S_HL_dot);
227  comp_P_LU_dot = DQmult(DQconj(S_LU_dot),P_w,S_LU)+DQmult(DQconj(S_LU),P_w,S_LU_dot);
228  comp_P_UB_dot = DQmult(DQconj(S_UB_dot),P_e,S_UB)+DQmult(DQconj(S_UB),P_e,S_UB_dot);
229  P_BS_dot_dq = DQmult(DQconj(S_tot),P_BS_0_dot_dq,S_tot)+...
230  DQmult(DQconj(S_LU_S_UB),comp_P_HL_dot,S_LU_S_UB)+...
231  DQmult(DQconj(S_UB),comp_P_LU_dot,S_LU_S_UB)+...
232  comp_P_UB_dot;
233  P_BS_dot = dquat2vel(P_BS_dot_dq);
234  end

236  disp(P_BS);

238  disp(P_BS_dot);

240  disp(orientation_tool_B);

242  disp(orientation_tool_B_dot);
243 end
