3 % ROTMATRIX2VFICK transforms a 3*3 rotation matrix into Fick coordinates
4 %
5 % [VFICK,VFICK2] = ROTMATRIX2FICK(ROTMATRIX) returns the Fick
6 % coordinates VFICK and VFICK2 [deg] corresponding to the rotation
7 % matrix ROTMATRIX.
8 % - ROTMATRIX is a 3*3 array or a 3*3*N tensor (in which case,
9 % ROTMATRIX(:,:,ii) is a 3*3 matrix representing the rotation
10 % matrix corresponding to rotation ii) where N is the number of
11 % rotations. ROTMATRIX can also be a 1*N cell and each cell
12 % component ROTMATRIX{1,ii} is the 3*3 rotation matrix of
13 % rotation ii.
14 % The following reference frame must be used for the rotation
15 % matrix! The positive rotation around x corresponds to a
16 % clockwise torsional rotation. The positive rotation around y
17 % corresponds to a downward rotation. The positive rotation
18 % about z corresponds to a leftward rotation.
19 % - VFICK (resp. VFICK2) is a 3*N array (column i represents
20 % rotation i). VFICK(:,i) can be written as VFICK = [H V T] where
21 % (from the subject's perspective):
22 % - H is the horizontal fick angle (positive when to the left)
23 % - V is the vertical fick angle (around the inter-aural axis).
24 % It is positive when down and is comprise between -90 and
25 % 90deg.
26 % - T is the torsional fick angle (around the line of signt). It
27 % is positive when clockwise.
28 % - VFICK2 is the second solution VFICK2 = [H2 V2 T2] and V2 is
29 % comprised between 90 and 270deg.
30 %
31 % Useful reference: T. Haslwanter (1995), Mathematics of three-dimensional
32 % eye rotations, Vision research, 35(12), 1727-39
33 %
34 % See also FICK2ROTMATRIX, DQUAT2ROTMATRIX, ROTMATRIX2DQUAT
35 
36 if nargout < 2 || nargout > 2
37  error('DualQuaternion:rotMatrix2fick:wrongNumberOfOutputs',...
38  'You specified %d outputs. It should be 2 (see documentation).',nargout);
39 end

42 isrotMcell = iscell(A);

44 if isrotMcell == 1

52  'Matrix size of rotation %d is %d*%d. It should be 3 and 3.',...
53  ii,sM(1),sM(2));
54  end
55  newA(:,:,ii) = curM;
56  end
57  A = newA;
58 else
59  sM = size(A);
60  ntensor = length(sM);
61  if ntensor < 2 || ntensor > 3
62  error('DualQuaternion:rotMatrix2fick:wrongsize',...
63  'The input is not a rotation matrix');

66  error('DualQuaternion:rotMatrix2fick:wrongsize',...
67  'First two dimensions of the tensor are %d and %d. It should be 3 and 3.',...
68  sM(1),sM(2));
69  end
70  end
71 end
72 % construction of the Fick coordinates
73 r31 = A(3,1,:); % = -sin(f)
74 V1 = asind(-r31); % V between [-90 90] deg (it could be between [90 and 270]
75 V2 = 180-V1; % alternative solution
76 
77 r11 = A(1,1,:);
78 r21 = A(2,1,:);
79 r32 = A(3,2,:);
80 r33 = A(3,3,:);
81 if min(abs(cosd(V1))) > 0
82  H1 = 180/pi*atan2(r21,r11);
83  T1 = 180/pi*atan2(r32,r33);
84  H2 = 180/pi*atan2(-r21,-r11);
85  T2 = 180/pi*atan2(-r32,-r33);
86 else

89 end
