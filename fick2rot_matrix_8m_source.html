1 function rotMatrix = fick2rotMatrix(vfick,varargin)
2 
3 % FICK2ROTMATRIX transforms Fick coordinates into a 3*3 rotation matrix
4 %
5 % ROTMATRIX = FICK2ROTMATRIX(VFICK) returns the rotation matrix ROTMATRIX
6 % corresponding to the Fick coordinates VFICK [deg].
7 % - VFICK is a 3-vector or a 3*N array (column i represents
8 % rotation i) where N is the number of rotations. The 3-D
9 % vector VFICK (or VFICK(:,i)) can be written as VFICK = [H V T]
10 % where (from the subject's perspective):
11 % - H is the horizontal fick angle (positive when to the left)
12 % - V is the vertical fick angle (around the inter-aural axis).
13 % It is positive when down.
14 % - T is the torsional fick angle (around the line of signt). It
15 % is positive when clockwise.
16 % - ROTMATRIX is a 3*3 array or a 3*3*N tensor (in which case,
17 % ROTMATRIX(:,:,ii) is a 3*3 matrix representing the rotation
18 % matrix corresponding to rotation ii.
19 %
20 % ROTMATRIX = FICK2ROTMATRIX(VFICK,OPTION) returns the rotation matrix
21 % ROTMATRIX under the form specified by the OPTION string:
22 % - 'tensor': ROTMATRIX is a 3*3*N tensor (the default)
23 % - 'cell': ROTMATRIX is a 1*N cell and each cell component
24 % ROTMATRIX{1,ii} is the 3*3 rotation matrix of rotation
25 % ii
26 %
27 % Useful reference: T. Haslwanter (1995), Mathematics of three-dimensional
28 % eye rotations, Vision research, 35(12), 1727-39
29 %
30 % See also ROTMATRIX2DQUAT, DQUAT2ROTMATRIX, ROTMATRIX2FICK
31 
32 sv = size(vfick);
33 if sv == [1 3], vfick = vfick'; sv = size(vfick); end
34 
35 % wrong format
36 if sv(1) ~= 3
37  error('DualQuaternion:fick2rotMatrix:wrongsize',...
38  '%d rows in the VFICK array. It should be 3.',sv(1));
39 end
40 vfick = vfick./180.*pi; % expressed in radians
41 n = sv(2);
42 
43 % rotation matrix components (in Fick convention)
44 t = vfick(1,:); % Horizontal angle : + when to the left (see Haslwanter95)
45 f = vfick(2,:); % Vertical angle: + when down
46 p = vfick(3,:); % Torsional angle: + when clockwise
47 
48 a11 = cos(t).*cos(f);
49 a22 = sin(t).*sin(f).*sin(p)+cos(t).*cos(p);
50 a33 = cos(f).*cos(p);
51 a23 = sin(t).*sin(f).*cos(p) - cos(t).*sin(p);
52 a32 = cos(f).*sin(p);
53 a31 = -sin(f);
54 a13 = cos(t).*sin(f).*cos(p) + sin(t).*sin(p);
55 a12 = cos(t).*sin(f).*sin(p)-sin(t).*cos(p);
56 a21 = sin(t).*cos(f);
57 
58 choice = 0;
59 soptargin = size(varargin);
60 if soptargin(2) > 1
61  error('DualQuaternion:fick2rotMatrix:toomanyInputs',...
62  'Too Many Inputs');
63 elseif soptargin(2) == 1
64  opt = varargin{1,1};
65  if strcmp(opt,'cell') % option is 'cell'
66  choice = 1;
67  end
68 end
69 
70 if choice == 0 % tensor
71  rotMatrix = zeros(3,3,n);
72  rotMatrix(1,1,:) = a11;
73  rotMatrix(1,2,:) = a12;
74  rotMatrix(1,3,:) = a13;
75  rotMatrix(2,1,:) = a21;
76  rotMatrix(2,2,:) = a22;
77  rotMatrix(2,3,:) = a23;
78  rotMatrix(3,1,:) = a31;
79  rotMatrix(3,2,:) = a32;
80  rotMatrix(3,3,:) = a33;
81 elseif choice == 1 % cell
82  rotMatrix = cell(1,n);
83  for ii=1:n
84  rotMatrix{1,ii} = [a11(ii) a12(ii) a13(ii);...
85  a21(ii) a22(ii) a23(ii);...
86  a31(ii) a32(ii) a33(ii)];
87  end
88 end
